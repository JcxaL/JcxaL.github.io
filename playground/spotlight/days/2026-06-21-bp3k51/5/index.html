3 · The Code

styles.css 
/* Step 1: tell the browser you support both modes */
html {
  color-scheme: light dark;
}

/* Step 2: use light-dark() anywhere */
body {
  background: light-dark(#f5f3ff, #07060f);
  color:       light-dark(#1a1030, #e8e0f8);
}

.card {
  background: light-dark(#ffffff, rgba(255,255,255,.05));
  border: 1.5px solid
    light-dark(rgba(0,0,0,.08), rgba(255,255,255,.08));
}

/* Step 3: force mode with color-scheme on :root */
html[data-mode=dark] {
  color-scheme: dark;
}

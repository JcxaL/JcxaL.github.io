CSS Ink Splash

Click any button — the ripple origin tracks exactly where you clicked.

/* Set ripple origin from click position */ el.addEventListener('click', e => { const r = el.getBoundingClientRect(); el.style.setProperty('--x', (e.clientX - r.left) + 'px'); el.style.setProperty('--y', (e.clientY - r.top) + 'px'); el.classList.remove('splashing'); void el.offsetWidth; /* force reflow to restart animation */ el.classList.add('splashing'); });
